# Pod Using ConfigMap Values (Environment Variables)
# pod-env.yaml
---
apiVersion: v1
kind: Pod
metadata:
  name: app-pod
spec:
  containers:
    - name: app
      image: nginx
      env:
        - name: APP_MODE #env variable inside container. 
          valueFrom:
            configMapKeyRef:
              name: app-config
              key: APP_MODE #key inside ConfigMap.data section
