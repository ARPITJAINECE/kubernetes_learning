a.
Example values:
username = admin
password = mypassword


b.
base64 encode them
echo -n admin | base64 -------> YWRtaW4=
echo -n mypassword | base64 ---------> bXlwYXNzd29yZA==


c.
secrets.yaml
---
apiVersion: v1
kind: Secret
metadata:
  name: db-secret
type: Opaque
data:
  DB_USERNAME: YWRtaW4=
  DB_PASSWORD: bXlwYXNzd29yZA==


d.
using secrets in pod
---
apiVersion: v1
kind: Pod
metadata:
  name: app-pod
spec:
  containers:
    - name: app
      image: myapp
      env:
        - name: DB_USERNAME
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: DB_USERNAME

        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: DB_PASSWORD


e.
what happens inside container : 

echo $DB_USERNAME  → admin
echo $DB_PASSWORD  → mypassword
